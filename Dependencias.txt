# Crear un nuevo proyecto con Vite el punto es para no crear uan carpeta extra
npm create vite@latest .  

# Instalar las dependencias de Node.js
npm install // NODE MODULES

# Instalar los íconos de Material-UI junto con sus dependencias
npm install @mui/icons-material @mui/material @emotion/styled @emotion/react // ICONS

# Instalar los componentes de Material-UI
npm install @mui/material @emotion/react @emotion/styled // COMPONENTES MATERIA

# Instalar Tailwind CSS y su integración con Vite
npm install tailwindcss @tailwindcss/vite // TAILWIND

# Instalar React Router DOM
npm install react-router-dom // ROUTER

# Instalar Headless UI (aún no utilizado)
npm install @headlessui/react // AÚN NO UTILIZADO
